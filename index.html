Setup - Header 

		<head>
			<script type="text/javascript" src="http://maps.googleapis.com/maps/api/js?sensor=true"></script> 
			<script type="text/javascript" src="http://ajax.googleapis.com/ajax/libs/jquery/1.6.2/jquery.min.js"></script>
			<script type="text/javascript" src="js/jquery-gMapTubeD.min.js"></script>		
		</head>
		
		//Optional - Allows MapTubeD Plugin to play nicely with other js libraries!
		<script type="text/javascript">
  			jQuery.noConflict();
		</script>
		
		//Inline CSS - You must, at minimum, give a height for the map container
		<style type="text/css">
			.map{
				width:  1030px;
				height: 400px;
			}
		</style>

Example 1 - Simple Map 

		<div id="simpleMap" class="map"></div>
		<script type="text/javascript">
			jQuery(document).ready(function () {
    				jQuery("#simpleMap").gMapTubeD({ descriptor: 'http://www.surveymapper.com/datafarm.svc/172/303/descriptor' });
    			});
    		</script>

Example 2 - Simple Map (with some options) 

  		<div id="simpleMapWithOptions" class="map"></div>
  		<script type="text/javascript">
			jQuery(document).ready(function () {
				jQuery("#simpleMapWithOptions").gMapTubeD({ 
					descriptor: 'http://bigdata.casa.ucl.ac.uk/2012/data/2012_tweets_desc.xml',
					latitude:   51.500143,
					longitude:  -0.126232,
					maptype:    'sat',
					zoom:	    10
				});
    			});
    		</script>

Example 3 - Multi Layered, Geocoded Map 

  		<div id="geocodeMap" class="map"></div>
  		<script type="text/javascript">
			jQuery(document).ready(function () {
				jQuery("#geocodeMap").gMapTubeD({
					descriptor: [
							'http://www.surveymapper.com/datafarm.svc/172/303/descriptor',
							'http://www.textal.org/maptube/textal_descriptor.xml'
						    ],
					address: 'London',
					zoom: 	 10,
					maptype: 'terrain'
				});	
    			});
    		</script>

Example 4 - Multi Layered, Advanced Options Map 

		<div id="advancedMultiLayered" class="map"></div>
		<script type="text/javascript">
			jQuery(document).ready(function () {
	
				jQuery("#advancedMultiLayered").gMapTubeD({
						descriptor: [
            					 {  layer: 'http://www.textal.org/maptube/textal_descriptor.xml', type: 'mtd', opacity: 0.3 },
            					  {  layer: 'http://www.surveymapper.com/datafarm.svc/172/303/descriptor',  type: 'smr', opacity: 0.7 }
           					],
           					address: 'UK',
						maptype: 'terrain',
						zoom: 5
				});		
			});			
		</script>

Example 5 - Styled Map 

		<div id="styledMap" class="map"></div>
		<script type="text/javascript">
			jQuery(document).ready(function () {
				var styleOptions = [
						{   	
						    featureType: 'all',
              					    stylers: [
               	 						{saturation: -100},
                						{gamma: 0.50},
		  						{invert_lightness: 'true'}
              		  			    	     ]
              		  			}, 
	     					{ featureType: "all",  elementType: "labels", stylers: [ { visibility: "off" } ] },
						{ featureType: "poi",  elementType: "all",    stylers: [ { visibility: "off" } ] },
						{ featureType: "road", elementType: "all",    stylers: [ { visibility: "off" } ] }
				 ];   
			
			
				jQuery("#styledMap").gMapTubeD({ 
					descriptor: 'http://bigdata.casa.ucl.ac.uk/2012/data/2012_tweets_ww_desc.xml',
					setStyle: { id:'Dark Night',  style: styleOptions },
					latitude: 	51.500143,
					longitude:  -0.126232,
					zoom:		5
				});
			});
		</script>
